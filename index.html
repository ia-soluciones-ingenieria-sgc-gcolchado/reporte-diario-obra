Conexión a Google Sheets

Como alojarás este archivo index.html en GitHub Pages, ingresa aquí la URL de tu Google Apps Script (Web App) para que los datos pasen de forma directa a tu hoja de cálculo.

✔ URL guardada en la memoria del celular

¿Cómo crear el Apps Script en Google Sheets?

  1. En tu Google Sheets ve a Extensiones > Apps Script.
  2. Reemplaza todo el código por este script directo:
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // Si la hoja está vacía, crear la fila de encabezados automáticamente if (sheet.getLastRow() === 0) { sheet.appendRow(["Fecha", "Proyecto", "Responsable", "Clima", "Personal Total", "Resumen Actividades", "Resumen Equipos", "Observaciones", "Firma", "Fecha Registro"]); sheet.getRange(1, 1, 1, 10).setBackground("#1e293b").setFontColor("#ffffff").setFontWeight("bold"); } // Formatear el resumen de actividades y equipos var actSummary = data.actividades.map(a => "• " + a.partida + " (" + a.cantidad + " " + a.unidad + ") [" + a.estado + "]").join("\n"); var eqSummary = data.equipos.map(eq => "• " + eq.nombre + " (" + eq.horas + "h) [" + eq.estado + "]").join("\n"); sheet.appendRow([ data.fecha, data.proyecto, data.responsable, data.clima.toUpperCase(), data.totalPersonal, actSummary, eqSummary, data.ocurrencias, data.firma ? "Firma Capturada" : "Sin firma", new Date().toLocaleString() ]); return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Fila insertada correctamente"})) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()})) .setMimeType(ContentService.MimeType.JSON); } }

3. Haz clic en Implementar > Nueva implementación, selecciona "Aplicación web", en acceso selecciona "Cualquier persona" y copia la URL resultante en el cuadro superior.
