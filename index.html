function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(["Fecha", "Proyecto", "Responsable", "Clima", "Personal Total", "Resumen Actividades", "Resumen Equipos", "Observaciones", "Firma", "Fecha Registro"]); sheet.getRange(1, 1, 1, 10).setBackground("#1e293b").setFontColor("#ffffff").setFontWeight("bold"); } var actSummary = data.actividades.map(a => "[" + a.codigo + "] " + a.partida + " (" + a.cantidad + " " + a.unidad + ") - " + a.estado).join("\n"); var eqSummary = data.equipos.map(eq => "• " + eq.nombre + " (" + eq.horas + "h) [" + eq.estado + "]").join("\n"); sheet.appendRow([ data.fecha, data.proyecto, data.responsable, data.clima.toUpperCase(), data.totalPersonal, actSummary, eqSummary, data.ocurrencias, data.firma ? "Firma Capturada" : "Sin firma", new Date().toLocaleString() ]); return ContentService.createTextOutput(JSON.stringify({"status": "success", "message": "Fila insertada correctamente"})) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
